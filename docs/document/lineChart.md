    [8, 30, 50, 82, 73, 84, 50],

];
const legendData = ['用水', '用电', '用工'];
const unit = ['吨', '千瓦时', '人'];
// 组合配置项
const chartOption = {
    xAxisData,
    seriesData,
    legendData,
    unit
};

onMounted(() => chartRef.value.renderChart());
</script>
